desc: small functions

@init

function front_width(F_width_coeff)
(
	F_M=(spl0+spl1)*0.5;
	F_S=(spl0-spl1)*0.5*F_width_coeff;

	spl0=F_M+F_S;
	spl1=F_M-F_S;
);


function rear_width(R_width_coeff)
(
	R_M=(spl4+spl5)*0.5;
	R_S=(spl4-spl5)*0.5*R_width_coeff;

	spl4=R_M+R_S;
	spl5=R_M-R_S;
);
